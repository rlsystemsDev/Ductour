## Changing Shapes And Animating Points

<Playground>
  <Component 
    initialState={{animatedPoint:[0,0]}} 
    didMount={(component) => {
      // This is a demo, in your code make sure to clearInterval on unmount
      setInterval(() => {
        component.setState({animatedPoint: [Math.random(), Math.random()]})
      }, 1500)
    }}
  >
    {(component) => {
      return <Chart 
        width={"600px"}
        height={"400px"}
        chartType="ScatterChart"
        loader={<div>Loading Chart</div>}
        data={[
          ['x', 'y'],
          component.state.animatedPoint,
          [4,0],[3.510330247561491,1.917702154416812],[2.161209223472559,3.365883939231586],[0.2829488066708116,3.989979946416218],[-1.6645873461885696,3.637189707302727],[-3.204574462187735,2.3938885764158258],[-3.9599699864017817,0.5644800322394689],[-3.7458267491631854,-1.4031329107584793],[-2.6145744834544478,-3.027209981231713],[-0.8431831977231188,-3.910120470660388],[1.134648741852905,-3.835697098652554],[2.83467909716504,-2.8221613022815677],[3.8406811466014643,-1.1176619927957034]
