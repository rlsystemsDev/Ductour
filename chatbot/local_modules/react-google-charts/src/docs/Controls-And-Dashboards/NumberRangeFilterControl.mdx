## Number Range Filter Control Example 1

<Playground>
  <Chart 
    width={400}
    height={300}
    chartType="BarChart"
    loader={<div>Loading Chart</div>}
    data={[
      ["Name", "Age"],
      ["Michael", 12],
      ["Elisa", 20],
      ["Robert", 7],
      ["John", 54],
      ["Jessica", 22],
      ["Aaron", 3],
      ["Margareth", 42],
      ["Miranda", 33]
    ]}
    options={{
      hAxis: { minValue: 0, maxValue: 60 },
      chartArea: { top: 0, right: 0, bottom: 0 }
    }}
    rootProps={{'data-testid': '4'}}
    chartPackages={["corechart", "controls"]}
    controls={[
      {
        controlType: "NumberRangeFilter",
        options: {
          filterColumnIndex: 1,
          minValue: 0,
          maxValue: 60
        }
      }
    ]}
  />
</Playground>



## Number Range Filter Control Example 2

<Playground>
  <Chart 
    width={"800px"}
    height={"400px"}
    chartType="PieChart"
    loader={<div>Loading Chart</div>}
    data={[
      ['Name', 'Gender', 'Age', 'Donuts eaten'],
      ['Michael' , 'Male', 12, 5],
      ['Elisa', 'Female', 20, 7],
      ['Robert', 'Male', 7, 3],
      ['John', 'Male', 54, 2],
      ['Jessica', 'Female', 22, 6],
      ['Aaron', 'Male', 3, 1],
      ['Margareth', 'Female', 42, 8]
    ]}
    rootProps={{'data-testid': '1'}}
    chartWrapperParams={{view: {columns: [0, 3]}}}
    chartPackages={["corechart", "controls"]}
    controls={[
      {
        controlType: "NumberRangeFilter",
        options: {
          'filterColumnLabel': 'Donuts eaten'
        },
        controlWrapperParams: {
          'state': {'lowValue': 2, 'highValue': 7}
        }
      }
    ]}
  />
</Playground>
